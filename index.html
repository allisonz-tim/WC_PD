function renderGroup() {

    const groupLetter = document.getElementById("groupSelect").value;

    const matches = tournamentData[groupLetter];

    

    let standings = {};

    let teamsSet = new Set();

    matches.forEach(m => { teamsSet.add(m.h); teamsSet.add(m.a); });

    let teamList = Array.from(teamsSet);

    

    teamList.forEach(t => standings[t] = { name: t, p: 0, w: 0, d: 0, l: 0, gf: 0, ga: 0, gd: 0 });


    matches.forEach(m => {

        standings[m.h].gf += m.hs; standings[m.h].ga += m.as;

        standings[m.a].gf += m.as; standings[m.a].ga += m.hs;

        if(m.hs > m.as) { standings[m.h].p += 3; standings[m.h].w++; standings[m.a].l++; }

        else if(m.hs < m.as) { standings[m.a].p += 3; standings[m.a].w++; standings[m.h].l++; }

        else { standings[m.h].p += 1; standings[m.a].p += 1; standings[m.h].d++; standings[m.a].d++; }

    });


    let sorted = Object.values(standings).sort((a,b) => {

        if(b.p !== a.p) return b.p - a.p;

        return (b.gf - b.ga) - (a.gf - a.ga);

    });


    let tableHtml = `<h2>📊 Group ${groupLetter} 官方預期積分榜</h2>

        <table>

            <tr><th>排名</th><th>球隊</th><th>勝</th><th>平</th><th>負</th><th>進/失球</th><th>淨勝球</th><th>積分</th><th>出線狀態</th></tr>`;

    sorted.forEach((t, i) => {

        t.gd = t.gf - t.ga;

        let status = i < 2 ? "<span class='q-direct'>直通 32 強</span>" : (i === 2 ? "<span class='q-wildcard'>爭取第三名</span>" : "小組淘汰");

        tableHtml += `<tr><td>${i+1}</td><td><strong>${t.name}</strong></td><td>${t.w}</td><td>${t.d}</td><td>${t.l}</td><td>${t.gf}/${t.ga}</td><td>${t.gd > 0 ? '+'+t.gd : t.gd}</td><td><strong>${t.p}</strong></td><td>${status}</td></tr>`;

    });

    tableHtml += `</table>`;


    let matchHtml = `<h2>🔥 Group ${groupLetter} 逐場分析 (含開賽時間)</h2>`;

    matches.forEach(m => {

        let tag = m.isReal ? "<span class='tag-real'>真實賽果</span>" : "<span class='tag-predict'>智能預測</span>";

        matchHtml += `

            <div class="match-card">

                <div class="match-header">

                    <span>📅 ${m.d} <span class="match-time-tag">⏰ ${m.t}</span> ｜ 🏟️ 觀眾: ${m.att.toLocaleString()}人</span> 

                    <span>${tag} ${m.prob ? '｜ ' + m.prob : ''}</span>

                </div>
